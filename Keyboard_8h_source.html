195  static QString convertLayout(const QString &text, const std::map<QChar, QChar> &charMap) {
196  QString convertedText = "";
197  for (int i = 0; i < text.size(); i++) {
198  if (charMap.find(text.at(i)) != charMap.end()) {
199  convertedText += charMap.at(text.at(i));

201  convertedText += text.at(i);

203  }

207  static Keyboard convertLayout(Keyboard &keyboard, const std::map<QChar, QChar> &charMap) {
208  Keyboard convertedKeyboard;
209  for (int i = 0; i < keyboard.getKeys().size(); i++) {
210  Key key = keyboard.getKeys()[i];
211  if (key.getType() == KeyType::REGULAR) {
212  key = key.withCharacters(convertLayout(key.getCharacters(), charMap));
213  }
214  convertedKeyboard.mKeys.push_back(key);
215  }
216  return convertedKeyboard;
217  }

219  static Keyboard convertLayout(Keyboard &keyboard, const QString &from, const QString &to) {
220  // create a QChar map from inputLayout to outputLayout
221  std::map<QChar, QChar> charMap;
222  for (int i = 0; i < from.size(); i++) {
223  charMap[from.at(i)] = to.at(i);
224  charMap[QChar(from.at(i)).toUpper()] = QChar(to.at(i)).toUpper();
225  }
226  return convertLayout(keyboard, charMap);

229  void loadSpecials(QString path = ":/specials.txt") {
230  QFile file(path);
231  file.open(QIODevice::ReadOnly);
232  if (!file.isOpen()) {
233  qDebug() << "Could not open specials file";
234  return;
235  }
236  /* File format :
237  a : à, â, ä, æ, ã, å, ā, ą, α (alpha)
238  b : ß, β (bêta)
239  c : ç, ć, č, ĉ, ċ, χ (chi)
240  */
241  std::map<QChar, QStringList> mSpecials;
242  while (!file.atEnd()) {
243  QString line = file.readLine();
244  QStringList parts = line.split(":");
245  if (parts.size() != 2) {
246  continue;
247  }
248  QString key = parts[0].trimmed();
249  QStringList values = parts[1].split(",");
250  for (int i = 0; i < values.size(); i++) {
251  values[i] = values[i].trimmed();
252  }
253  mSpecials[key.at(0)] = values;
254  }
255 
256  // apply specials to keys
257  for (int i = 0; i < mKeys.size(); i++) {
258  Key &key = mKeys[i];
259  if (key.getType() != KeyType::REGULAR) {
260  continue;
261  }
262  QString characters = key.getCharacters();
263  for (int j = 0; j < characters.size(); j++) {
264  if (mSpecials.find(characters.at(j)) != mSpecials.end()) {
265  key.setSpecials(j, mSpecials[characters.at(j)]);
266  }
267  }
268  }
269  }
