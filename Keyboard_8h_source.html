179  static inline Keyboard importKeyboard(const QString &name, const QString &layout) {

193  static QString convertLayout(const QString &text, const std::map<QChar, QChar> &charMap) {
194  QString convertedText = "";
195  for (int i = 0; i < text.size(); i++) {
196  if (charMap.find(text.at(i)) != charMap.end()) {
197  convertedText += charMap.at(text.at(i));
198  } else {
199  convertedText += text.at(i);
200  }
201  }
202  return convertedText;
203  }

205  static Keyboard convertLayout(Keyboard &keyboard, const std::map<QChar, QChar> &charMap) {
206  Keyboard convertedKeyboard;
207  for (int i = 0; i < keyboard.getKeys().size(); i++) {
208  Key key = keyboard.getKeys()[i];
209  if (key.getType() == KeyType::REGULAR) {
210  key = key.withCharacters(convertLayout(key.getCharacters(), charMap));
211  }
212  convertedKeyboard.mKeys.push_back(key);
213  }
214  return convertedKeyboard;
215  }

217  static Keyboard convertLayout(Keyboard &keyboard, const QString &from, const QString &to) {
218  // create a QChar map from inputLayout to outputLayout
219  std::map<QChar, QChar> charMap;
220  for (int i = 0; i < from.size(); i++) {
221  charMap[from.at(i)] = to.at(i);
222  charMap[QChar(from.at(i)).toUpper()] = QChar(to.at(i)).toUpper();
223  }
224  return convertLayout(keyboard, charMap);
225  }
