200  static inline Key deserialize(QFile &file) {

203  file.read((char*)(&key.mType), sizeof(KeyType));

205  file.read((char*)(&key.mYSpan), sizeof(float));
206  file.read((char*)(&key.mX), sizeof(float));
207  file.read((char*)(&key.mY), sizeof(float));
208 
209  int mCharactersSize;
210  file.read((char*)(&mCharactersSize), sizeof(int));
211  if (mCharactersSize > 0) {
212  char* mCharactersBuffer = new char[mCharactersSize];
213  file.read(mCharactersBuffer, mCharactersSize);
214  key.mCharacters = QString(mCharactersBuffer);
215  delete[] mCharactersBuffer;
216  }
217  return key;
